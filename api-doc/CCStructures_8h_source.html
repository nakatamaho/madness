774  friend std::vector<std::pair<double,CCConvolutionOperator>> combine(const CCConvolutionOperator& left, const CCConvolutionOperator& right) {
775  MADNESS_CHECK(can_combine(left,right));
776  MADNESS_CHECK(left.world.id()==right.world.id());
777  auto [type,param]=combine_OT(left,right);
778  std::vector<std::pair<double,CCConvolutionOperator>> result;
779  if (type==OT_FG12) {
780  // fg = (1 - exp(-gamma r12)) / r12 = 1/r12 - exp(-gamma r12)/r12 = coulomb - bsh
781 
782  // coulombfit return 1/r
783  // we need 1/(2 gamma) 1/r
784  result.push_back(std::make_pair(1.0/(2.0*param.gamma),CCConvolutionOperator(left.world, OT_G12, param)));
785 
786  // bshfit returns 1/(4 pi) exp(-gamma r)/r
787  // we need 1/(2 gamma) exp(-gamma r)/r
788  const double factor = 4.0 * constants::pi /(2.0*param.gamma);
789  result.push_back(std::make_pair(-factor,CCConvolutionOperator(left.world, OT_BSH, param)));
790  } else if (type==OT_F212) {
791 // we use the slater operator which is S = e^(-y*r12), y=gamma
792 // the f12 operator is: 1/2y*(1-e^(-y*r12)) = 1/2y*(1-S)
793 // so the squared f12 operator is: f*f = 1/(4*y*y)(1-2S+S*S), S*S = S(2y) = e(-2y*r12)
794 // we have then: <xy|f*f|xy> = 1/(4*y*y)*(<xy|xy> - 2*<xy|S|xy> + <xy|SS|xy>)
795 // we have then: <xy|f*f|xy> =(<xy|f12|xy> - 1/(4*y*y)*2*<xy|S|xy>
796  MADNESS_CHECK(left.parameters.gamma==right.parameters.gamma);
797  const double prefactor = 1.0 / (4.0 * param.gamma); // Slater has no 1/(2 gamma) per se.
798  Parameters param2=param;
799  param2.gamma*=2.0;
800  result.push_back(std::make_pair(1.0*prefactor,CCConvolutionOperator(left.world, OT_ONE, param)));
801  result.push_back(std::make_pair(-2.0*prefactor,CCConvolutionOperator(left.world, OT_SLATER, left.parameters)));
802  result.push_back(std::make_pair(1.0*prefactor,CCConvolutionOperator(left.world, OT_SLATER, param2)));
803  }
804  return result;
805  }
