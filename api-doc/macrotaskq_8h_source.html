1 /**
2  \file macrotaskq.h
3  \brief Declares the \c macrotaskq and MacroTaskBase classes
4  \ingroup mra
5 
6  A MacroTaskq executes tasks on World objects, e.g. differentiation of a function or other
7  arithmetic. Complex algorithms can be implemented.
8 
9  The universe world is split into subworlds, each of them executing macrotasks of the task queue.
10  This improves locality and speedups for large number of compute nodes, by reducing communications
11  within worlds.
12 
13  The user defines a macrotask (an example is found in test_vectormacrotask.cc), the tasks are
14  lightweight and carry only bookkeeping information, actual input and output are stored in a
15  cloud (see cloud.h)
16 
17  The user-defined macrotask is derived from MacroTaskIntermediate and must implement the run()
18  method. A heterogeneous task queue is possible.
19 
20  TODO: priority q

325  /// store all input to the cloud, create output Function<T,NDIM> in the universe,
326  /// create the batched task and shove it into the taskq. Possibly execute the taskq.
327  template<typename ... Ts>
328  resultT operator()(const Ts &... args) {
329 
330  const bool immediate_execution = (not taskq_ptr);
331  if (not taskq_ptr) taskq_ptr.reset(new MacroTaskQ(world, world.size()));
332 
333  auto argtuple = std::tie(args...);
334  static_assert(std::is_same<decltype(argtuple), argtupleT>::value, "type or number of arguments incorrect");
335 
336  // partition the argument vector into batches
337  auto partitioner=task.partitioner;
338  if (not partitioner) partitioner.reset(new MacroTaskPartitioner);
339  partitioner->set_nsubworld(world.size());
340  partitionT partition = partitioner->partition_tasks(argtuple);
341 
342  // store input and output: output being a pointer to a universe function (vector)
343  recordlistT inputrecords = taskq_ptr->cloud.store(world, argtuple);
344  auto[outputrecords, result] =prepare_output(taskq_ptr->cloud, argtuple);
345 
346  // create tasks and add them to the taskq
347  MacroTaskBase::taskqT vtask;
348  for (const auto& batch_prio : partition) {
349  vtask.push_back(
350  std::shared_ptr<MacroTaskBase>(new MacroTaskInternal(task, batch_prio, inputrecords, outputrecords)));
351  }
352  taskq_ptr->add_tasks(vtask);
353 
354  if (immediate_execution) taskq_ptr->run_all(vtask);
355 
356  return result;
357  }

364  /// prepare the output of the macrotask: Function<T,NDIM> must be created in the universe
365  std::pair<recordlistT, resultT> prepare_output(Cloud &cloud, const argtupleT &argtuple) {
366  static_assert(is_madness_function<resultT>::value || is_madness_function_vector<resultT>::value);
367  resultT result = task.allocator(world, argtuple);
368  recordlistT outputrecords;
369  if constexpr (is_madness_function<resultT>::value) {
370  outputrecords += cloud.store(world, result.get_impl().get()); // store pointer to FunctionImpl
371  } else if constexpr (is_vector<resultT>::value) {
372  outputrecords += cloud.store(world, get_impl(result));
373  } else {
374  MADNESS_EXCEPTION("\n\n unknown result type in prepare_input ", 1);
375  }

420 
421  const argtupleT argtuple = cloud.load<argtupleT>(subworld, inputrecords);
422  const argtupleT batched_argtuple = task.batch.template copy_input_batch(argtuple);
423
