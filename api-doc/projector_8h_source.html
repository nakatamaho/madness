31 /// simple projector class
32 
33  /// use this class to project a function or a set of functions on
34  /// another space of function. The projector can handle different sets of
35  /// functions for the bra and the ket space, e.g. in case of regularized
36  /// orbitals: |f> <-> <f|R^2
37  template<typename T, std::size_t NDIM>
38  class Projector : public ProjectorBase {
39 
40  typedef Function<T,NDIM> funcT;
41  typedef std::vector<funcT> vecfuncT;
42 
43  /// the space onto which the test functions will be projected: |ket>
44  std::vector<Function<T,NDIM> > mo_ket_;
45  /// the dual space onto which the test functions will be projected: <bra|
46  std::vector<Function<T,NDIM> > mo_bra_;
47 
48  public:
49 
50  Projector() : mo_ket_(vecfuncT()), mo_bra_(vecfuncT()) {}
51 
52  /// simple constructor with only one orbital to project
53 
54  /// bra and ket spaces are symmetric
55  Projector(const Function<T,NDIM>& mo) : mo_ket_(vecfuncT(1,mo)),
56  mo_bra_(vecfuncT(1,mo)) {}
57

59

61  Projector(const funcT& bra, const funcT& ket) : mo_ket_(vecfuncT(1,ket))
62  , mo_bra_(vecfuncT(1,bra)) {}

70

73  mo_bra_(bra) {}

77  /// project f on p:
78 
79  /// \f[
80  /// | result > = \sum_p | p > <p|f>
81  /// \f]
82  /// @param[in] f the function to be projected
83  /// @return the projection of f on the space of p
84  funcT operator()(const funcT& f) const {
85  return this->operator()(vecfuncT(1,f)).front();
86  }
87 
88  /// project f on p:
89 
90  /// \f[
91  /// | result > = \sum_p | p > <p|f>
92  /// \f]
93  /// @param[in] f the vector of functions to be projected
94  /// @return the projection of f on the space of p
95  vecfuncT operator()(const vecfuncT& f) const {
96  if (f.size()==0) return vecfuncT();
97  World& world=f[0].world();
98  Tensor<T> ovlp=matrix_inner(world,mo_bra_,f);
99  vecfuncT result=transform(world,mo_ket_,ovlp,true);
100  truncate(world,result);
101  return result;
102  }
103 
104  /// apply 3D Projector to one particle of a 6D function
105  /// \f[
106  /// |result> = \sum_p |p(particle)> <p(particle)|f(1,2)>_{particle}
107  /// \f]
108  /// @param[in] f the 6D function to be projected
109  /// @param[in] the particle that is projected (1 or 2)
110  /// @return the projected function
111  real_function_6d operator()(const real_function_6d& f, const size_t particle) const {
112  real_function_6d result = real_factory_6d(f.world());
113  MADNESS_ASSERT(particle == 1 or particle == 2);
114  for (size_t i = 0; i < mo_ket_.size(); i++) {
115  real_function_3d tmp1 = mo_ket_[i];
116  real_function_3d tmp2 = f.project_out(mo_bra_[i], particle - 1);
117  real_function_6d tmp12;
118  if (particle == 1) {
119  tmp12 = CompositeFactory<double, 6, 3>(f.world()).particle1(copy(tmp1)).particle2(copy(tmp2));
120  tmp12.fill_tree();
121  } else {
122  tmp12 = CompositeFactory<double, 6, 3>(f.world()).particle1(copy(tmp2)).particle2(copy(tmp1));
123  tmp12.fill_tree();
124  }
125  result += tmp12;
126  }
127  return result;
128  }

270  /// notation of the equations follows
271  /// J. Chem. Phys., vol. 139, no. 11, p. 114106, 2013.
272  Function<T,2*NDIM> operator()(const Function<T,2*NDIM>& f) const {
273 
274  // simple and it works for higher accuracies, but might be
275  // imprecise for lower accuracies
276 // return (f-O1(f)-O2(f)+O1(O2(f))).truncate().reduce_rank();
277 
278  const double thresh=FunctionDefaults<2*NDIM>::get_thresh();
279  const double tight_thresh=FunctionDefaults<2*NDIM>::get_thresh()*0.1;
280 
281  // Eq. (A9): g_kl = < k(1) l(2) | f(1,2) >
282  // note no (kl) symmetry here!
283  Tensor<double> g_kl(bra1_.size(),bra2_.size());
284  for (size_t k=0; k<bra1_.size(); ++k) {
285  for (size_t l=0; l<bra2_.size(); ++l) {
286  Function<T,2*NDIM> kl=CompositeFactory<T,2*NDIM,NDIM>(world)
287  .particle1(copy(bra1_[k])).particle2(copy(bra2_[l]));
288  g_kl(k,l)=inner(f,kl);
289  }
290  }
291 // if (world.rank()==0) {print(g_kl);};
292 
293  // Eq. (A12)
294  // project out the mainly first particle: O1 (1 - 1/2 O2)
295  Function<T,2*NDIM> r1=FunctionFactory<T,2*NDIM>(world).thresh(tight_thresh);
296  for (size_t k=0; k<bra1_.size(); ++k) {
297 
298  // Eq. (A10)
299  Function<T,NDIM> h2=f.project_out(bra1_[k],0);
300 
301  // Eq. (A12)
302  for (size_t l=0; l<ket2_.size(); ++l) {
303  h2-=0.5*g_kl(k,l)*ket2_[l];
304  }
305 
306  // Eq. (A7), second term rhs
307  // the hartree product tends to be inaccurate; tighten threshold
308  FunctionDefaults<2*NDIM>::set_thresh(tight_thresh);
309  r1=(r1+hartree_product(ket1_[k],h2));
310  FunctionDefaults<2*NDIM>::set_thresh(thresh);
311  r1.set_thresh(thresh);
312  r1.print_size("r1"+stringify(k));
313  }
314 
315  // project out the mainly second particle: O2 (1 - 1/2 O1)
316  Function<T,2*NDIM> r2=FunctionFactory<T,2*NDIM>(world).thresh(tight_thresh);
317  for (size_t l=0; l<ket2_.size(); ++l) {
318 
319  // Eq. (A11)
320  Function<T,NDIM> h1=f.project_out(bra2_[l],1);
321 
322  // Eq. (A13)
323  for (size_t k=0; k<ket1_.size(); ++k) {
324  h1-=0.5*g_kl(k,l)*ket1_[k]; // ordering g(k,l) is correct
325  }
326 
327  // Eq. (A7), third term rhs
328  // the hartree product tends to be inaccurate; tighten threshold
329  FunctionDefaults<2*NDIM>::set_thresh(tight_thresh);
330  r2=(r2+hartree_product(h1,ket2_[l]));
331  r2.set_thresh(thresh);
332  FunctionDefaults<2*NDIM>::set_thresh(thresh);
333  r2.print_size("r2"+stringify(l));
334  }
335  FunctionDefaults<2*NDIM>::set_thresh(tight_thresh);
336  Function<T,2*NDIM> result=(f-r1-r2).truncate().reduce_rank();
337  FunctionDefaults<2*NDIM>::set_thresh(thresh);
338
